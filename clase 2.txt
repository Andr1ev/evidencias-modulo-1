Ejercicio compra tienda virtual

entradas:
	-navegador
	-productos
	-celular
	-datos usuario

proceso:
	-ingresar al sitio web de la compra
	-logearse en el sitio
	-realizar busqueda de los productos
	-seleccionar productos
	-agregar productos al carro de compras
	-ingresar al carro de compras
	-confirmar productos
	-confirmar datos del usuario
	-realizar compra
	-confirmar que la compra fue exitosa
	-salir del sitio

salida:
	-compra de los productos
